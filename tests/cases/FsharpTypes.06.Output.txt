namespace Domain
open Protokeep.FsharpTypes
type TrafficLight =
    | Unknown = 0
    | Red = 1
    | Yellow = 2
    | Green = 3
type Crossroad = {
    Id : int
    Light : TrafficLight
    Street1 : string
    Street2 : string
}
with
    static member MakeKey (id': int, light': TrafficLight) =
        Key.Items [Key.Value (id'.ToString()); Key.Value ((int light').ToString())]
    interface IEntity with
        member x.Key = Crossroad.MakeKey (x.Id, x.Light)
